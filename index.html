NetworkInteractionTool.cs
C# · Unity Netcode for GameObjects NetworkBehaviour + ServerRpc + ClientRpc

A multiplayer interaction tool that fires raycasts from a held object. The owner client gets immediate local feedback while the server validates hits and broadcasts visual effects to all other clients via ClientRpc. Uses a queue to flush pending effects each frame without allocs.

using System.Collections.Generic;
using Unity.Netcode;
using UnityEngine;

public class NetworkInteractionTool : NetworkBehaviour
{
    [SerializeField] private Transform      muzzle;
    [SerializeField] private float          range      = 18f;
    [SerializeField] private float          fireRate   = 0.15f;
    [SerializeField] private LayerMask      hitMask;
    [SerializeField] private LineRenderer   scanBeam;
    [SerializeField] private ParticleSystem hitParticles;

    private float _nextFireTime;

    private readonly Queue<(Vector3 from, Vector3 to, bool hit)> _effectQueue = new();

    private void Update()
    {
        FlushEffectQueue();

        if (!IsOwner || !Input.GetButton("Fire1") || Time.time < _nextFireTime)
            return;

        _nextFireTime = Time.time + fireRate;

        Vector3 origin    = muzzle.position;
        Vector3 direction = muzzle.forward;
        bool    didHit    = Physics.Raycast(origin, direction, out RaycastHit local, range, hitMask);

        _effectQueue.Enqueue((origin, didHit ? local.point : origin + direction * range, didHit));

        SubmitRaycastServerRpc(origin, direction);
    }

    [ServerRpc]
    private void SubmitRaycastServerRpc(Vector3 origin, Vector3 direction)
    {
        bool hit = Physics.Raycast(origin, direction, out RaycastHit info, range, hitMask);

        if (hit && info.collider.TryGetComponent(out IInteractable interactable))
            interactable.Interact(OwnerClientId);

        Vector3 endpoint = hit ? info.point : origin + direction * range;
        BroadcastEffectsClientRpc(origin, endpoint, hit);
    }

    [ClientRpc]
    private void BroadcastEffectsClientRpc(Vector3 origin, Vector3 endpoint, bool hit)
    {
        if (IsOwner) return;
        _effectQueue.Enqueue((origin, endpoint, hit));
    }

    private void FlushEffectQueue()
    {
        while (_effectQueue.Count > 0)
        {
            var (from, to, hit) = _effectQueue.Dequeue();
            ShowBeam(from, to);
            if (hit) ShowHitEffect(to);
        }
    }

    private void ShowBeam(Vector3 from, Vector3 to)
    {
        scanBeam.enabled = true;
        scanBeam.SetPositions(new[] { from, to });
        CancelInvoke(nameof(HideBeam));
        Invoke(nameof(HideBeam), 0.08f);
    }

    private void HideBeam()               => scanBeam.enabled = false;
    private void ShowHitEffect(Vector3 at) => hitParticles.transform.position = at;
}
ObjectPool.cs
C# · Unity · Generic Systems Stack-based O(1) pooler

A generic, GC-friendly object pool that pre-allocates a reusable set of Component instances. ProjectilePool shows the concrete usage pattern with a singleton facade — so callers never touch the pool API directly.

using System.Collections.Generic;
using UnityEngine;

public sealed class ObjectPool<T> where T : Component
{
    private readonly T        _prefab;
    private readonly Transform _root;
    private readonly Stack<T>  _pool = new();

    public int ActiveCount { get; private set; }

    public ObjectPool(T prefab, int warmup = 0, Transform root = null)
    {
        _prefab = prefab;
        _root   = root;

        for (int i = 0; i < warmup; i++)
            _pool.Push(Spawn());
    }

    public T Get(Vector3 position, Quaternion rotation)
    {
        T obj = _pool.Count > 0 ? _pool.Pop() : Spawn();
        obj.transform.SetPositionAndRotation(position, rotation);
        obj.gameObject.SetActive(true);
        ActiveCount++;
        return obj;
    }

    public void Return(T obj)
    {
        obj.gameObject.SetActive(false);
        _pool.Push(obj);
        ActiveCount--;
    }

    public void Prewarm(int count)
    {
        for (int i = 0; i < count; i++)
            _pool.Push(Spawn());
    }

    private T Spawn()
    {
        T obj = Object.Instantiate(_prefab, _root);
        obj.gameObject.SetActive(false);
        return obj;
    }
}

public class ProjectilePool : MonoBehaviour
{
    [SerializeField] private Projectile prefab;
    [SerializeField] private int        warmupSize = 20;

    private ObjectPool<Projectile> _pool;

    public static ProjectilePool Instance { get; private set; }

    private void Awake()
    {
        Instance = this;
        _pool    = new ObjectPool<Projectile>(prefab, warmupSize, transform);
    }

    public Projectile Get(Vector3 position, Quaternion rotation)
        => _pool.Get(position, rotation);

    public void Return(Projectile p) => _pool.Return(p);
}
AbilityController.cs
C# · Unity · ScriptableObject Architecture Data-driven ability system

A data-driven ability system using ScriptableObjects for ability definitions. AbilityContext is a readonly struct (zero-GC context passing), AbilitySlot owns runtime state while the definition stays immutable. The controller resolves aim point via camera raycast and feeds it into each slot each frame.

using UnityEngine;

[CreateAssetMenu(menuName = "Abilities/Definition")]
public abstract class AbilityDefinition : ScriptableObject
{
    [field: SerializeField] public string AbilityName { get; private set; }
    [field: SerializeField] public float  Cooldown    { get; private set; } = 1f;
    [field: SerializeField] public float  EnergyCost  { get; private set; } = 10f;
    [field: SerializeField] public Sprite Icon        { get; private set; }

    public abstract void Execute(in AbilityContext ctx);

    public bool CanActivate(float energy) => energy >= EnergyCost;
}

public readonly struct AbilityContext
{
    public readonly Transform  Origin;
    public readonly Vector3    AimPoint;
    public readonly LayerMask  TargetMask;

    public AbilityContext(Transform origin, Vector3 aimPoint, LayerMask mask)
    {
        Origin     = origin;
        AimPoint   = aimPoint;
        TargetMask = mask;
    }
}

public class AbilitySlot
{
    public AbilityDefinition Definition        { get; }
    public float             CooldownRemaining { get; private set; }
    public bool              IsReady           => CooldownRemaining <= 0f;

    public AbilitySlot(AbilityDefinition definition) => Definition = definition;

    public void Tick(float delta) => CooldownRemaining = Mathf.Max(0f, CooldownRemaining - delta);

    public bool TryActivate(in AbilityContext ctx, ref float energy)
    {
        if (!IsReady || !Definition.CanActivate(energy)) return false;

        energy            -= Definition.EnergyCost;
        CooldownRemaining  = Definition.Cooldown;
        Definition.Execute(ctx);
        return true;
    }
}

public class AbilityController : MonoBehaviour
{
    [SerializeField] private AbilityDefinition[] slotDefinitions;
    [SerializeField] private LayerMask           aimMask;
    [SerializeField] private float               maxEnergy   = 100f;
    [SerializeField] private float               energyRegen = 8f;

    private static readonly KeyCode[] SlotKeys = { KeyCode.Q, KeyCode.E, KeyCode.R, KeyCode.F };

    private AbilitySlot[] _slots;
    private float         _energy;
    private Camera        _cam;

    private void Awake()
    {
        _cam    = Camera.main;
        _energy = maxEnergy;
        _slots  = new AbilitySlot[slotDefinitions.Length];

        for (int i = 0; i < slotDefinitions.Length; i++)
            _slots[i] = new AbilitySlot(slotDefinitions[i]);
    }

    private void Update()
    {
        _energy = Mathf.Min(_energy + energyRegen * Time.deltaTime, maxEnergy);

        AbilityContext ctx = BuildContext();

        for (int i = 0; i < _slots.Length; i++)
        {
            _slots[i].Tick(Time.deltaTime);

            if (i < SlotKeys.Length && Input.GetKeyDown(SlotKeys[i]))
                _slots[i].TryActivate(ctx, ref _energy);
        }
    }

    private AbilityContext BuildContext()
    {
        Ray     ray      = _cam.ScreenPointToRay(Input.mousePosition);
        bool    didHit   = Physics.Raycast(ray, out RaycastHit hit, 100f, aimMask);
        Vector3 aimPoint = didHit ? hit.point : ray.origin + ray.direction * 50f;

        return new AbilityContext(transform, aimPoint, aimMask);
    }
}
